impl Solution {
    pub fn plus_one(digits: Vec<i32>) -> Vec<i32> {
        match digits
            .iter()
            .enumerate()
            .rev()
            .skip_while(|(i, &x)| 9 == x)
            .next()
        {
            Some((i, &a)) => [
                digits[..i]
                    .iter()
                    .map(|x| x.clone())
                    .collect::<Vec<i32>>()
                    .as_slice(),
                vec![digits[i] + 1].as_slice(),

            .concat(),

        }
